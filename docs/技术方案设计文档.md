### 2.1 ER关系图（文字版）

```

users ──1:N── ai_text_records

training_items ──1:N── practice_records

growth_goal_config ──独立配置表（单例，4条记录）

recommend_config ──独立配置表（3条记录）

admin_users ──1:N── operation_logs
push_records ──独立记录表
```

### 2.2 完整建表语句

```sql
-- ============================================================
-- 数据库创建
-- ============================================================
CREATE DATABASE IF NOT EXISTS eloquence_camp
  DEFAULT CHARACTER SET utf8mb4
  DEFAULT COLLATE utf8mb4_unicode_ci;
USE eloquence_camp;

-- ============================================================
-- 1. 用户表
-- ============================================================
CREATE TABLE users (
    id              INT UNSIGNED    AUTO_INCREMENT PRIMARY KEY,
    openid          VARCHAR(64)     NOT NULL UNIQUE        COMMENT '微信openid',
    unionid         VARCHAR(64)     DEFAULT NULL           COMMENT '微信unionid(用于跨应用识别)',
    nickname        VARCHAR(64)     DEFAULT ''             COMMENT '微信昵称',
    avatar_url      VARCHAR(512)    DEFAULT ''             COMMENT '微信头像URL',
    total_days      INT UNSIGNED    DEFAULT 0              COMMENT '累计打卡天数',
    continuous_days INT UNSIGNED    DEFAULT 0              COMMENT '连续打卡天数',
    total_practice_minutes INT UNSIGNED DEFAULT 0          COMMENT '累计练习总时长(分钟)',
    ability_score   JSON            DEFAULT NULL           COMMENT '5维能力评分JSON: {pronunciation:80,fluency:75,...}',
    growth_level    VARCHAR(20)     DEFAULT 'newbie'       COMMENT '成长等级: newbie/beginner/advanced/expert/master',
    created_at      DATETIME        DEFAULT CURRENT_TIMESTAMP,
    updated_at      DATETIME        DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_openid (openid),
    INDEX idx_growth_level (growth_level),
    INDEX idx_created_at (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';

-- ============================================================
-- 2. 用户权益配额表
-- ============================================================
CREATE TABLE user_quota (
    id              INT UNSIGNED    AUTO_INCREMENT PRIMARY KEY,
    user_id         INT UNSIGNED    NOT NULL UNIQUE         COMMENT '关联用户ID',
    daily_ai_quota  INT UNSIGNED    DEFAULT 3               COMMENT '每日AI文案基础次数',
    daily_ai_used   INT UNSIGNED    DEFAULT 0               COMMENT '今日已用AI次数',
    extra_quota     INT UNSIGNED    DEFAULT 0               COMMENT '额外赠送次数（永久增加）',
    last_reset_date DATE            DEFAULT NULL            COMMENT '上次次数重置日期，非今日则重置daily_ai_used',
    created_at      DATETIME        DEFAULT CURRENT_TIMESTAMP,
    updated_at      DATETIME        DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户权益配额表';

-- ============================================================
-- 3. 训练题表
-- ============================================================
CREATE TABLE training_items (
    id              INT UNSIGNED    AUTO_INCREMENT PRIMARY KEY,
    category        VARCHAR(30)     NOT NULL                COMMENT '大类: basic/speech/livestream/improv',
    sub_category    VARCHAR(30)     DEFAULT NULL            COMMENT '子类: 如pronunciation/self_intro/opening',
    title           VARCHAR(100)    NOT NULL                COMMENT '训练题标题',
    difficulty      TINYINT         DEFAULT 1               COMMENT '难度: 1=⭐ 2=⭐⭐ 3=⭐⭐⭐',
    sample_text     TEXT            NOT NULL                COMMENT '范文文本(最长3000字)',
    sample_audio_url VARCHAR(512)   DEFAULT NULL            COMMENT '范本音频OSS地址',
    tags            JSON            DEFAULT NULL            COMMENT '标签JSON数组: ["竞聘","面试","结构化"]',
    status          VARCHAR(20)     DEFAULT 'online'        COMMENT '状态: online/offline',
    practice_count  INT UNSIGNED    DEFAULT 0               COMMENT '累计练习次数(冗余缓存)',
    sort_order      INT             DEFAULT 0               COMMENT '排序号，越小越靠前',
    created_at      DATETIME        DEFAULT CURRENT_TIMESTAMP,
    updated_at      DATETIME        DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_category (category),
    INDEX idx_status_sort (status, sort_order),
    INDEX idx_difficulty (difficulty)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='训练题表';

-- ============================================================
-- 4. 练习记录表
-- ============================================================
CREATE TABLE practice_records (
    id              INT UNSIGNED    AUTO_INCREMENT PRIMARY KEY,
    user_id         INT UNSIGNED    NOT NULL                COMMENT '用户ID',
    training_item_id INT UNSIGNED   DEFAULT NULL            COMMENT '训练题ID(自由练习可为空)',
    audio_url       VARCHAR(512)    NOT NULL                COMMENT '用户录音OSS地址',
    duration        INT UNSIGNED    DEFAULT 0               COMMENT '录音时长(秒)',
    ai_score        TINYINT UNSIGNED DEFAULT NULL           COMMENT 'AI综合评分(0-100)',
    dimension_scores JSON           DEFAULT NULL            COMMENT '5维评分JSON',
    ai_feedback     TEXT            DEFAULT NULL            COMMENT 'AI改进建议文字',
    source          VARCHAR(20)     DEFAULT 'free_practice' COMMENT '来源: checkin_task/free_practice/ai_import',
    created_at      DATETIME        DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_user_created (user_id, created_at DESC),
    INDEX idx_training_item (training_item_id),
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (training_item_id) REFERENCES training_items(id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='练习记录表';

-- ============================================================
-- 5. 打卡记录表
-- ============================================================
CREATE TABLE checkin_records (
    id              INT UNSIGNED    AUTO_INCREMENT PRIMARY KEY,
    user_id         INT UNSIGNED    NOT NULL                COMMENT '用户ID',
    task_date       DATE            NOT NULL                COMMENT '打卡日期',
    status          VARCHAR(20)     DEFAULT 'completed'     COMMENT '状态: completed/makeup/rest',
    completed_tasks JSON            DEFAULT NULL            COMMENT '完成的任务详情JSON: [{task_id,score,duration}]',
    created_at      DATETIME        DEFAULT CURRENT_TIMESTAMP,
    updated_at      DATETIME        DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UNIQUE KEY uk_user_date (user_id, task_date),
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='打卡记录表';

-- ============================================================
-- 6. 每日任务配置表（单例配置，后台编辑）
-- ============================================================
CREATE TABLE daily_task_config (
    id              INT UNSIGNED    AUTO_INCREMENT PRIMARY KEY,
    task_index      TINYINT         NOT NULL UNIQUE         COMMENT '任务序号: 1/2/3',
    title           VARCHAR(64)     NOT NULL                COMMENT '任务标题: 跟读朗读',
    subtitle        VARCHAR(128)    DEFAULT ''              COMMENT '副标题',
    min_duration    INT UNSIGNED    DEFAULT 60              COMMENT '最低录音时长(秒)',
    source_type     VARCHAR(20)     DEFAULT 'random'        COMMENT '范本来源: random/fixed',
    source_category VARCHAR(30)     DEFAULT NULL            COMMENT '随机抽取范围-分类',
    source_difficulty_min TINYINT   DEFAULT 1               COMMENT '随机抽取-最低难度',
    source_difficulty_max TINYINT   DEFAULT 3               COMMENT '随机抽取-最高难度',
    is_active       TINYINT(1)      DEFAULT 1               COMMENT '是否启用',
    created_at      DATETIME        DEFAULT CURRENT_TIMESTAMP,
    updated_at      DATETIME        DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='每日任务配置表';

-- ============================================================
-- 7. 成长目标配置表
-- ============================================================
CREATE TABLE growth_goal_config (
    id              INT UNSIGNED    AUTO_INCREMENT PRIMARY KEY,
    goal_level      VARCHAR(20)     NOT NULL UNIQUE         COMMENT '目标等级: beginner/advanced/expert/master',
    goal_name       VARCHAR(64)     NOT NULL                COMMENT '目标名称: 7天入门',
    required_days   INT UNSIGNED    NOT NULL                COMMENT '累计打卡天数要求',
    reward_level    VARCHAR(30)     DEFAULT NULL            COMMENT '解锁素材等级',
    reward_extra_ai INT UNSIGNED    DEFAULT 0               COMMENT '额外AI次数奖励',
    badge_icon      VARCHAR(10)     DEFAULT ''              COMMENT '徽章emoji',
    badge_name      VARCHAR(30)     DEFAULT ''              COMMENT '徽章名称',
    is_active       TINYINT(1)      DEFAULT 1               COMMENT '是否启用',
    created_at      DATETIME        DEFAULT CURRENT_TIMESTAMP,
    updated_at      DATETIME        DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='成长目标配置表';

-- ============================================================
-- 8. AI文案生成记录表
-- ============================================================
CREATE TABLE ai_text_records (
    id              INT UNSIGNED    AUTO_INCREMENT PRIMARY KEY,
    user_id         INT UNSIGNED    NOT NULL                COMMENT '用户ID',
    scene_type      VARCHAR(30)     NOT NULL                COMMENT '场景: speech/short_video/livestream/opening',
    input_params    JSON            NOT NULL                COMMENT '用户输入参数JSON',
    generated_content TEXT          NOT NULL                COMMENT '生成的文案',
```

## 4. 关键业务流程时序

### 4.1 用户完成一次打卡练习（完整链路）

```
小程序端                    Flask后端                   阿里云服务
   │                          │                          │
   │  ①GET /checkin/today     │                          │
   │ ───────────────────────→ │ 查询daily_task_config    │
   │ ←─────────────────────── │ + 随机抽取训练题          │
   │  返回3个任务              │                          │
   │                          │                          │
   │  ②点击任务2"开始练习"     │                          │
   │  进入录音页               │                          │
   │                          │                          │
   │  ③播放范本音频            │                          │
   │ ─────────────────────────────────────────────────→ │
   │ ←───────────────────────────────────────────────── │
   │  从OSS加载范本音频        │                          │
   │                          │                          │
   │  ④开始录音→录制完成       │                          │
   │                          │                          │
   │  ⑤POST /training/practice│                          │
   │  (上传录音文件)           │                          │
   │ ───────────────────────→ │ 上传录音→OSS             │
   │                          │ ───────────────────────→ │
   │                          │ ←── 返回audio_url        │
   │                          │                          │
   │                          │ 调用语音评测API           │
   │                          │ ───────────────────────→ │
   │                          │ ←── 发音/流利度/完整度    │
   │                          │                          │
   │                          │ 调用Qwen3-Flash           │
   │                          │ (分析内容/逻辑/表现力)    │
   │                          │ ───────────────────────→ │
   │                          │ ←── 反馈文字              │
   │                          │                          │
   │                          │ 写入practice_records     │
   │ ←─────────────────────── │                          │
   │  返回评分+反馈            │                          │
   │                          │                          │
   │  ⑥POST /checkin/complete │                          │
   │  -task (task_index=2)    │                          │
   │ ───────────────────────→ │ 校验+标记任务完成         │
   │                          │ 检查3个是否全部完成       │
   │ ←─────────────────────── │                          │
   │                          │                          │
   │  ⑦(如果全部完成)          │                          │
   │                          │ 写入checkin_records      │
   │                          │ 更新users打卡计数         │
   │                          │ 检查成长目标              │
   │                          │ 赠送AI次数                │
   │                          │                          │
   │                          │ 触发微信订阅消息推送      │
   │                          │ ───────────────────────→ │
   │                          │                          │ 微信服务通知
   │                          │                          │ ──→ 用户微信
   │ ←─────────────────────── │                          │
   │  打卡成功! +奖励信息       │                          │
```
